43  self.var.DeltaTSnow = 0.9674 * loadmap('ElevationStD') * loadmap('TemperatureLapseRate')
44 
45  # Difference between (average) air temperature at average elevation of
46  # pixel and centers of upper- and lower elevation zones [deg C]
47  # ElevationStD: Standard Deviation of the DEM from Bodis (2009)
48  # 0.9674: Quantile of the normal distribution: u(0,833)=0.9674
49  # to split the pixel in 3 equal parts.

51  # day of the year to degrees: 360/365.25 = 0.9856
52 
53  self.var.IceDayDegrees = 1.915
54  # days of summer (15th June-15th Sept.) to degree: 180/(259-165)
55  self.var.SnowSeason = loadmap('SnowSeasonAdj') * 0.5
56  # default value of range of seasonal melt factor is set to 1

81  SeasSnowMeltCoef = self.var.SnowSeason * np.sin(math.radians((self.var.CalendarDay - 81)
82  * self.var.SnowDayDegrees)) + self.var.SnowMeltCoef
83 
84  # SeasSnowMeltCoef = SnowSeason * sin((CalendarDay-81)* SnowDayDegrees) + SnowMeltCoef;
85 
86  # sinus shaped function between the
87  # annual minimum (December 21st) and annual maximum (June 21st)
88  # SummerSeason = ifthenelse(self.var.CalendarDay > 165,np.sin((self.var.CalendarDay-165)* self.var.IceDayDegrees ),scalar(0.0))
89  # SummerSeason = ifthenelse(self.var.CalendarDay > 259,0.0,SummerSeason)
90  if (self.var.CalendarDay > 165) and (self.var.CalendarDay < 260):
91  SummerSeason = np.sin(math.radians((self.var.CalendarDay - 165) * self.var.IceDayDegrees))
92  else:
93  SummerSeason = 0.0
94 
95  self.var.Snow = globals.inZero.copy()
96  self.var.Rain = globals.inZero.copy()
97  self.var.SnowMelt = globals.inZero.copy()
98  self.var.SnowCover = globals.inZero.copy()
99 
100  for i in xrange(3):
101  TavgS = self.var.Tavg + self.var.DeltaTSnow * (i - 1)
102  # Temperature at center of each zone (temperature at zone B equals Tavg)
103  # i=0 -> highest zone
104  # i=2 -> lower zone
105  SnowS = np.where(TavgS < self.var.TempSnow, self.var.SnowFactor * self.var.Precipitation, globals.inZero)
106  # Precipitation is assumed to be snow if daily average temperature is below TempSnow
107  # Snow is multiplied by correction factor to account for undercatch of
108  # snow precipitation (which is common)
109  RainS = np.where(TavgS >= self.var.TempSnow, self.var.Precipitation, globals.inZero)
110  # if it's snowing then no rain
111  SnowMeltS = (TavgS - self.var.TempMelt) * SeasSnowMeltCoef * (1 + 0.01 * RainS) * self.var.DtDay
112 
113  if i < 2:
114  IceMeltS = self.var.Tavg * 7.0 * self.var.DtDay * SummerSeason
115  # if i = 0 and 1 -> higher and middle zone
116  else:
117  IceMeltS = TavgS * 7.0 * self.var.DtDay * SummerSeason
118 
119  SnowMeltS = np.maximum(np.minimum(SnowMeltS + IceMeltS, self.var.SnowCoverS[i]), globals.inZero)
120  self.var.SnowCoverS[i] = self.var.SnowCoverS[i] + SnowS - SnowMeltS
121  self.var.Snow += SnowS
122  self.var.Rain += RainS
123  self.var.SnowMelt += SnowMeltS
124  self.var.SnowCover += self.var.SnowCoverS[i]
125 
126  self.var.Snow /= 3
127  self.var.Rain /= 3
128  self.var.SnowMelt /= 3
129  self.var.SnowCover /= 3
130 
131  self.var.TotalPrecipitation += self.var.Snow + self.var.Rain
